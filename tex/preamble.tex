\definecolor{lstbg}{RGB}{248,249,251}
\definecolor{lstcomment}{RGB}{100,110,120}
\definecolor{lstkeyword}{RGB}{0,77,153}
\definecolor{lststring}{RGB}{163,21,21}

\lstdefinelanguage{bash}{
    keywords={if,then,else,fi,for,while,do,done,in,function,case,esac,echo,exit,export},
    sensitive=true,
    morecomment=[l]{\#},
    morestring=[b]",
    morestring=[b]'
}

\lstdefinelanguage{cuda}{
    language=C++,
    morekeywords={
            __global__,__host__,__device__,__shared__,__constant__,__managed__,
            __syncthreads,threadIdx,blockIdx,blockDim,gridDim,warpSize,
            cudaMalloc,cudaFree,cudaMemcpy,cudaMemcpyAsync,cudaMemset,
            cudaStreamCreate,cudaStreamDestroy,cudaDeviceSynchronize
        },
    sensitive=true
}

\lstdefinelanguage{pseudocode}{
    keywords={
            algorithm,procedure,function,begin,end,if,then,else,for,while,repeat,until,
            return,input,output,true,false,break,continue
        },
    sensitive=false,
    morecomment=[l]{//},
    morestring=[b]"
}

    stepnumber=1,
    numbersep=8pt,
    showstringspaces=false,

}
